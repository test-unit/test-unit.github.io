def []=(id, scheme_or_spec)

    scheme = scheme_or_spec
  else
    scheme = new(scheme_or_spec)
  end
  @@schemes[id.to_s] = scheme
end

def available_colors
  case ENV["COLORTERM"]
  when "gnome-terminal"
    256
  else
    case ENV["TERM"]
    when /-256color\z/
      256
    else
      8
    end
  end
end
